CREATE TABLE RPS_CLERK(
CLERK_ID VARCHAR2(10) CONSTRAINT CLERK_PK PRIMARY KEY,
PASSWORD VARCHAR2(10) CONSTRAINT CLERK_NN_PS NOT NULL,
CLERK_NAME VARCHAR2(20)  
);


CREATE TABLE RPS_REGISTER(
USER_ID VARCHAR2(10) CONSTRAINT USER_PK PRIMARY KEY,
PASSWORD VARCHAR2(10)  CONSTRAINT USER_NN_PS NOT NULL,
USER_NAME VARCHAR2(30) CONSTRAINT USER_NN_UN NOT NULL,
MOBILE_NO NUMBER(13) CONSTRAINT USER_NN_MN NOT NULL,
ADDRESS VARCHAR(60) CONSTRAINT USER_NN_AD NOT NULL
);

CREATE TABLE RPS_PARCEL(
PRR_NO NUMBER(10) CONSTRAINT PARCEL_PRR_NO PRIMARY KEY,
CLERK_NAME VARCHAR2(20),
PARCEL_DATE DATE DEFAULT SYSDATE,
TRAIN_NO VARCHAR2(10),
SOURCE VARCHAR2(40) CONSTRAINT PARCEL_NN_STN NOT NULL,
DESTINATION VARCHAR2(40) CONSTRAINT PARCEL_NN_DTN NOT NULL,
ITEMTYPE VARCHAR2(40) CONSTRAINT PARCEL_NN_IT NOT NULL,
ITEMQUANTITY NUMBER(2) CONSTRAINT PARCEL_NN_IQ NOT NULL,
WEIGHT NUMBER(10) CONSTRAINT PARCEL_NN_W NOT NULL,
DISTANCE NUMBER(6) CONSTRAINT PARCEL_CK_DT CHECK(DISTANCE<10000),
PRICE NUMBER(6) CONSTRAINT PARCEL_NN_P NOT NULL,
SENDER_NAME VARCHAR2(30) CONSTRAINT PARCEL_NN_SN NOT NULL,
SENDER_MOBILE_NO VARCHAR2(13) CONSTRAINT PARCEL_NN_SMN NOT NULL,
SENDER_ADDRESS VARCHAR(60) CONSTRAINT PARCEL_NN_SAD NOT NULL,
RECIPIENT_NAME VARCHAR2(30) CONSTRAINT PARCEL_NN_RN NOT NULL,
RECIPIENT_MOBILE_NO VARCHAR2(13) CONSTRAINT PARCEL_NN_RMN NOT NULL,
RECIPIENT_ADDRESS VARCHAR(60) CONSTRAINT PARCEL_NN_RAD NOT NULL
);

CREATE TABLE RPS_PARCELSTATUS(
PRR_NO NUMBER(10),
CLERK_NAME VARCHAR2(20),
TRAIN_NO VARCHAR2(10),
PARCEL_STATUS VARCHAR2(50) CONSTRAINT PARCEL_STATUS_NN_PS NOT NULL,
CONSTRAINT PS_FK FOREIGN KEY(PRR_NO) REFERENCES RPS_PARCEL(PRR_NO)
);

CREATE SEQUENCE RPS_PRR
START WITH 1201
INCREMENT BY 10
NOCYCLE;
